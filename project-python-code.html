Structura proiectului

Un setup simplu, dar corect pentru un proiect de forecasting

Exemplul de mai jos presupune că ai un fișier cu trafic zilnic. Poate fi Excel sau CSV. În cod, facem preprocesare, generăm variabile de timp, antrenăm un model de regresie și producem grafice pentru perioada de test și pentru următoarele 30 de zile.

Pași de lucru

1

Pregătești datele

Un fișier cu coloane precum date și visits. Ideal: o înregistrare pe zi.

2

Construiești feature-uri de timp

Zi din săptămână, lună, weekend, lag-uri, medii mobile și trend în timp.

3

Împarți datele cronologic

Nu amesteci seriile temporale. Folosești primele observații pentru train și ultimele pentru test.

4

Antrenezi modelul

În exemplu folosim RandomForestRegressor, fiind robust și ușor de pornit.

5

Evaluezi și faci forecast

Calculezi MAE, RMSE, R² și generezi predicții pentru viitor cu grafice separate.

3. Script principal — forecast_web_traffic.py 
import os
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

from sklearn.ensemble import RandomForestRegressor
from sklearn.metrics import mean_absolute_error, mean_squared_error, r2_score


# =========================================
# CONFIG
# =========================================
DATA_PATH = "data/website_traffic.csv"   # sau "data/website_traffic.xlsx"
OUTPUT_DIR = "outputs"
TEST_SIZE = 0.2
FORECAST_DAYS = 30
RANDOM_STATE = 42


# =========================================
# LOAD DATA
# =========================================
def load_data(path):
    ext = os.path.splitext(path)[1].lower()

    if ext == ".csv":
        df = pd.read_csv(path)
    elif ext in [".xlsx", ".xls"]:
        df = pd.read_excel(path)
    else:
        raise ValueError("Format fișier neacceptat. Folosește CSV sau Excel.")

    if "date" not in df.columns or "visits" not in df.columns:
        raise ValueError("Fișierul trebuie să conțină coloanele: 'date' și 'visits'.")

    df["date"] = pd.to_datetime(df["date"])
    df = df.sort_values("date").reset_index(drop=True)

    # Eliminare duplicate pe dată, păstrăm ultima valoare
    df = df.drop_duplicates(subset=["date"], keep="last")

    # Asigurăm frecvență zilnică
    full_range = pd.date_range(start=df["date"].min(), end=df["date"].max(), freq="D")
    df = df.set_index("date").reindex(full_range).rename_axis("date").reset_index()

    # Completăm valorile lipsă prin interpolare
    df["visits"] = df["visits"].interpolate(method="linear")
    df["visits"] = df["visits"].fillna(method="bfill").fillna(method="ffill")

    return df


# =========================================
# FEATURE ENGINEERING
# =========================================
def create_features(df):
    df = df.copy()

    df["day_of_week"] = df["date"].dt.dayofweek
    df["day_of_month"] = df["date"].dt.day
    df["month"] = df["date"].dt.month
    df["week_of_year"] = df["date"].dt.isocalendar().week.astype(int)
    df["is_weekend"] = df["day_of_week"].isin([5, 6]).astype(int)
    df["day_index"] = (df["date"] - df["date"].min()).dt.days

    # lag features
    df["lag_1"] = df["visits"].shift(1)
    df["lag_7"] = df["visits"].shift(7)
    df["lag_14"] = df["visits"].shift(14)

    # rolling features
    df["rolling_mean_7"] = df["visits"].shift(1).rolling(window=7).mean()
    df["rolling_mean_14"] = df["visits"].shift(1).rolling(window=14).mean()

    # opțional: trend neliniar
    df["day_index_sq"] = df["day_index"] ** 2

    df = df.dropna().reset_index(drop=True)
    return df


# =========================================
# TRAIN / TEST SPLIT FOR TIME SERIES
# =========================================
def time_split(df, test_size=0.2):
    split_index = int(len(df) * (1 - test_size))
    train_df = df.iloc[:split_index].copy()
    test_df = df.iloc[split_index:].copy()
    return train_df, test_df


# =========================================
# TRAIN MODEL
# =========================================
def train_model(train_df, feature_cols, target_col="visits"):
    X_train = train_df[feature_cols]
    y_train = train_df[target_col]

    model = RandomForestRegressor(
        n_estimators=300,
        max_depth=10,
        min_samples_split=5,
        min_samples_leaf=2,
        random_state=RANDOM_STATE
    )

    model.fit(X_train, y_train)
    return model


# =========================================
# EVALUATION
# =========================================
def evaluate_model(model, test_df, feature_cols, target_col="visits"):
    X_test = test_df[feature_cols]
    y_test = test_df[target_col]

    preds = model.predict(X_test)

    mae = mean_absolute_error(y_test, preds)
    rmse = np.sqrt(mean_squared_error(y_test, preds))
    r2 = r2_score(y_test, preds)

    result_df = test_df[["date", target_col]].copy()
    result_df["predicted_visits"] = preds

    return mae, rmse, r2, result_df


# =========================================
# FUTURE FORECAST
# =========================================
def forecast_future(model, historical_df, feature_cols, forecast_days=30):
    future_predictions = []
    df_hist = historical_df.copy()

    for _ in range(forecast_days):
        next_date = df_hist["date"].max() + pd.Timedelta(days=1)

        next_row = {
            "date": next_date
        }

        temp_df = pd.concat([df_hist, pd.DataFrame([next_row])], ignore_index=True)

        temp_df["day_of_week"] = temp_df["date"].dt.dayofweek
        temp_df["day_of_month"] = temp_df["date"].dt.day
        temp_df["month"] = temp_df["date"].dt.month
        temp_df["week_of_year"] = temp_df["date"].dt.isocalendar().week.astype(int)
        temp_df["is_weekend"] = temp_df["day_of_week"].isin([5, 6]).astype(int)
        temp_df["day_index"] = (temp_df["date"] - temp_df["date"].min()).dt.days
        temp_df["day_index_sq"] = temp_df["day_index"] ** 2

        # lag-uri bazate pe istoricul deja existent + predicțiile anterioare
        temp_df["lag_1"] = temp_df["visits"].shift(1)
        temp_df["lag_7"] = temp_df["visits"].shift(7)
        temp_df["lag_14"] = temp_df["visits"].shift(14)

        temp_df["rolling_mean_7"] = temp_df["visits"].shift(1).rolling(window=7).mean()
        temp_df["rolling_mean_14"] = temp_df["visits"].shift(1).rolling(window=14).mean()

        last_row = temp_df.iloc[-1:].copy()

        # dacă lipsesc valori în primele zile, completăm din istoricul disponibil
        for col in ["lag_1", "lag_7", "lag_14", "rolling_mean_7", "rolling_mean_14"]:
            if pd.isna(last_row.iloc[0][col]):
                last_row[col] = df_hist["visits"].mean()

        pred = model.predict(last_row[feature_cols])[0]

        new_row = {
            "date": next_date,
            "visits": pred
        }

        future_predictions.append(new_row)
        df_hist = pd.concat([df_hist, pd.DataFrame([new_row])], ignore_index=True)

    future_df = pd.DataFrame(future_predictions)
    return future_df


# =========================================
# PLOTS
# =========================================
def plot_test_predictions(result_df, output_path):
    plt.figure(figsize=(14, 6))
    plt.plot(result_df["date"], result_df["visits"], label="Valori reale", linewidth=2)
    plt.plot(result_df["date"], result_df["predicted_visits"], label="Predicții", linewidth=2)
    plt.title("Trafic web: valori reale vs predicții (set test)")
    plt.xlabel("Data")
    plt.ylabel("Vizite")
    plt.legend()
    plt.grid(True, alpha=0.3)
    plt.tight_layout()
    plt.savefig(output_path, dpi=150)
    plt.close()


def plot_future_forecast(original_df, future_df, output_path):
    plt.figure(figsize=(14, 6))

    tail_df = original_df.tail(60)

    plt.plot(tail_df["date"], tail_df["visits"], label="Istoric recent", linewidth=2)
    plt.plot(future_df["date"], future_df["visits"], label="Forecast următoarele 30 zile", linewidth=2)

    plt.title("Predicția traficului web pentru perioada viitoare")
    plt.xlabel("Data")
    plt.ylabel("Vizite")
    plt.legend()
    plt.grid(True, alpha=0.3)
    plt.tight_layout()
    plt.savefig(output_path, dpi=150)
    plt.close()


# =========================================
# MAIN
# =========================================
def main():
    os.makedirs(OUTPUT_DIR, exist_ok=True)

    print("1. Încărcare date...")
    raw_df = load_data(DATA_PATH)

    print("2. Feature engineering...")
    model_df = create_features(raw_df)

    feature_cols = [
        "day_of_week",
        "day_of_month",
        "month",
        "week_of_year",
        "is_weekend",
        "day_index",
        "day_index_sq",
        "lag_1",
        "lag_7",
        "lag_14",
        "rolling_mean_7",
        "rolling_mean_14"
    ]

    print("3. Split train/test...")
    train_df, test_df = time_split(model_df, test_size=TEST_SIZE)

    print("4. Antrenare model...")
    model = train_model(train_df, feature_cols)

    print("5. Evaluare...")
    mae, rmse, r2, result_df = evaluate_model(model, test_df, feature_cols)

    print(f"MAE: {mae:.2f}")
    print(f"RMSE: {rmse:.2f}")
    print(f"R2: {r2:.4f}")

    result_df.to_csv(os.path.join(OUTPUT_DIR, "predictions_test.csv"), index=False)

    print("6. Forecast viitor...")
    future_df = forecast_future(
        model=model,
        historical_df=raw_df.copy(),
        feature_cols=feature_cols,
        forecast_days=FORECAST_DAYS
    )

    future_df.to_csv(os.path.join(OUTPUT_DIR, "forecast_next_30_days.csv"), index=False)

    print("7. Generare grafice...")
    plot_test_predictions(
        result_df,
        os.path.join(OUTPUT_DIR, "actual_vs_pred.png")
    )

    plot_future_forecast(
        raw_df,
        future_df,
        os.path.join(OUTPUT_DIR, "future_forecast.png")
    )

    print("Gata. Fișierele au fost salvate în folderul 'outputs'.")


if __name__ == "__main__":
    main()

Îmbunătățiri utile

Cum faci proiectul mai bun și mai realist

Exemplul de mai sus este foarte bun pentru început, dar în practică poți crește acuratețea dacă adaugi surse externe și modele suplimentare. Important este să păstrezi disciplina unei serii temporale reale.

Adaugă variabile externe

Include campanii de marketing, cheltuieli ads, sărbători, trafic social, poziții SEO sau newsletter sends. Traficul web este influențat rar doar de timp.

Compară mai multe modele

Testează și XGBoost, LightGBM, Prophet sau SARIMA. Uneori modelele clasice de time series bat modelele de ML generaliste.

Folosește validare temporală

În loc de un singur split train/test, poți aplica walk-forward validation. Asta îți dă o evaluare mai serioasă pentru forecasting real.

Format recomandat pentru un dataset mai bun 
date,visits,ad_spend,email_campaign,is_holiday,social_clicks

Observații importante

A

Ai nevoie de destule date

Pentru rezultate decente, încearcă să ai măcar 6–12 luni de date zilnice.

B

Nu trage concluzii doar din R²

Verifică și graficele, erorile pe perioade speciale și comportamentul în weekend sau în campanii.

C

Forecast-ul pe termen lung devine mai incert

Cu cât prezici mai departe, cu atât modelele bazate pe lag-uri acumulează eroare.
